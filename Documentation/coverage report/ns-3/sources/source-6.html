335 	 * averageColor returns the avgColor of inputted BufferedImage bi.
336 	 * @param bi
337 	 * @return
338 	 */
339 	public static Color averageColor(BufferedImage bi) {
340 		int step = 5;
341 
342 		int sampled = 0;
343 		long sumr = 0, sumg = 0, sumb = 0;
344 		for (int x = 0; x < bi.getWidth(); x++) {
345 			for (int y = 0; y < bi.getHeight(); y++) {
346 				if (x % step == 0 && y % step == 0) {
347 					Color pixel = new Color(bi.getRGB(x, y));
348 					sumr += pixel.getRed();
349 					sumg += pixel.getGreen();
350 					sumb += pixel.getBlue();
351 					sampled++;
352 				}
353 			}
354 		}
355 		//int dim = bi.getWidth()*bi.getHeight();
356 		return new Color(Math.round(sumr / sampled), Math.round(sumg / sampled), Math.round(sumb / sampled));
357 	}

359 	public static Color getContrastColor(Color color) {
360 		double y = (299 * color.getRed() + 587 * color.getGreen() + 114 * color.getBlue()) / 1000;
361 		return y >= 128 ? Color.black : Color.white;
