152 		switch (State) {
153 			//Search by Title
154 			case "Title":
155 				populateSearchList(ml.searchForMovie(input));
156 				break;
157 			case "Actor":
158 				populateSearchList(ml.filterByActor(input));
159 				ml.clearFilteredList();
160 				break;
161 			case "Director":
162 				populateSearchList(ml.filterByDirector(input));
163 				ml.clearFilteredList();
164 				break;
165 			//Default case is search by title
166 			default:
167 				populateSearchList(ml.searchForAny(input));
168 				break;
